#include <stdio.h>
#include <stdlib.h>

int main() {

    int n, i;
    int mas;
    int total;
    float *calif;
    float suma = 0;
    float promedio;

    // 1. Pedir cuantos estudiantes
    printf("Cuantos estudiantes deseas registrar? ");
    scanf("%d", &n);

    // 2. Reservar memoria con malloc
    calif = (float *)malloc(n * sizeof(float));

    // Verificar error
    if (calif == NULL) {
        printf("Error al asignar memoria\n");
        return 1;
    }

    // 3. Capturar calificaciones
    for (i = 0; i < n; i++) {
        printf("Calificacion del estudiante %d: ", i + 1);
        scanf("%f", &calif[i]);
        suma += calif[i];
    }

    total = n;

    // Calcular promedio inicial
    promedio = suma / total;
    printf("Promedio inicial: %.2f\n", promedio);

    // 5. Preguntar si quiere agregar mas
    printf("Deseas agregar mas estudiantes? (cantidad, 0 para no): ");
    scanf("%d", &mas);

    if (mas > 0) {

        // usar realloc
        calif = (float *)realloc(calif, (n + mas) * sizeof(float));

        if (calif == NULL) {
            printf("Error al reasignar memoria\n");
            return 1;
        }

        // capturar nuevas calificaciones
        for (i = n; i < n + mas; i++) {
            printf("Calificacion del estudiante %d: ", i + 1);
            scanf("%f", &calif[i]);
            suma += calif[i];
        }

        total = n + mas;
    }

    // 4 y 6 mostrar todo
    printf("\n=== Calificaciones registradas ===\n");

    for (i = 0; i < total; i++) {
        printf("Estudiante %d: %.2f\n", i + 1, calif[i]);
    }

    promedio = suma / total;

    printf("Promedio general: %.2f\n", promedio);

    // 7 liberar memoria
    free(calif);

    printf("Memoria liberada correctamente\n");

    return 0;
}
